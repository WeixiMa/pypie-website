@op
def softmax(x: Tensor[float][[m, n]]) -> Tensor[float][[m, n]]:
    x_exp = x.exp()
    sums = x_exp.sum(1).reshape([m, 1])
    return x_exp / sums

def cross_entropy(pred: Tensor[float][[m, n]], indices: Tensor[int][[m]]) -> float:
    probs = softmax(pred)
    labels = [row[idx] for row, idx in zip(probs, indices)]
    return -labels.log().mean()
